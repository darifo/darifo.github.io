斐波拉契数列

  1,1,2,3,5,8,13....

现在我们想求得第i位的菲波拉契数是多少，直接上代码：


```java
package com.company.darifo.solutions;

/*
 * 斐波拉切数列
 * 求第N位斐波拉切值 todo N太大本类不适用，因为int类型溢出，需要使用BigInteger类实现运算存储，算法思想是不变的
 * */
public class Fibonacci {

    public static void main(String[] args) {
        long ta = System.currentTimeMillis();
        System.out.println(fib(40));
        long tb = System.currentTimeMillis();
        System.out.println(fibWithMemo(40));
        long tc = System.currentTimeMillis();
        System.out.println(fibIteration(40));
        long td = System.currentTimeMillis();

        System.out.println(ta);
        System.out.println(tb);
        System.out.println(tc);
        System.out.println(td);
    }

    /*
     * todo 暴力递归 时间复杂度：O（2^n） 空间复杂度：O（1）
     * */
    static int fib(int N) {
        if (N == 1 || N == 2)
            return 1;
        return fib(N - 1) + fib(N - 2);
    }

    /*
     * todo 带备忘录的递归 时间复杂度：O（n）空间复杂度：O（n）
     * */
    static int fibWithMemo(int N) {
        if (N < 0)
            return 0;
        int[] memo = new int[N + 1];
        return helper(memo, N);
    }

    static int helper(int[] memo, int n) {
        if (n == 1 || n == 2)
            return 1;
        if (memo[n] == 0)
            memo[n] = helper(memo, n - 1) + helper(memo, n - 2);
        return memo[n];
    }

    /*
    * todo 迭代法 时间复杂度：O（N） 空间复杂度：O（1）
    * */
    static int fibIteration(int N){
        if (N==1 || N==2)
            return 1;
        int a =1 , b = 1;
        int sum = a + b;
        for (int i=3;i<N;i++){
            a = b;
            b = sum;
            sum = a+b;
        }
        return sum;
    }

}
```
